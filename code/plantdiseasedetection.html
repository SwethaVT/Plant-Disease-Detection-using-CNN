def convert_image_to_array(image_dir):
     try:
         image = cv2.imread(image_dir)
         if image is not None :
             image = cv2.resize(image, default_image_size)   
             return img_to_array(image)
         else :
             return np.array([])
     except Exception as e:
         print(f"Error : {e}")
         return None
